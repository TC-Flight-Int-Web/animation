setTimeout

      function click() {
        // 代码1
        setTimeout(function() {
          // 200毫秒后执行的代码
        }, 200);
        // 代码2
      }

setInterval

      function click() {
        // 代码1...
        setInterval(function() {
        // process ...
        }, 200);
        // 代码2
      }
